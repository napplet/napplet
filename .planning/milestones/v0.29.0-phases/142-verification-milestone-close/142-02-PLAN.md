    Structure:
    ```typescript
    import { describe, it, expect } from 'vitest';
    import { createHash } from 'node:crypto';

    /**
     * NUB-CONNECT canonical connect:origins aggregateHash fold.
     * Byte-identical to the vite-plugin implementation at
     * packages/vite-plugin/src/index.ts (Phase 138-02). Both copies
     * MUST agree on the spec conformance fixture digest.
     *
     * Procedure per NUB-CONNECT §Canonical Fold:
     *   1. Lowercase each origin (no-op if already lowercase)
     *   2. ASCII-ascending sort
     *   3. LF-join ('\n'), no trailing newline
     *   4. UTF-8 encode
     *   5. SHA-256
     *   6. Lowercase hex
     */
    function foldConnectOrigins(origins: readonly string[]): { digest: string; byteLength: number } {
      const lowercased = origins.map((o) => o.toLowerCase());
      const sorted = [...lowercased].sort();
      const joined = sorted.join('\n');
      const bytes = Buffer.from(joined, 'utf8');
      const digest = createHash('sha256').update(bytes).digest('hex');
      return { digest, byteLength: bytes.length };
    }

    // NUB-CONNECT §Conformance Fixture (spec-locked):
    const FIXTURE_ORIGINS = [
      'https://api.example.com',
      'https://xn--caf-dma.example.com',
      'wss://events.example.com',
    ];
    const EXPECTED_DIGEST = 'cc7c1b1903fb23ecb909d2427e1dccd7d398a5c63dd65160edb0bb8b231aa742';
    const EXPECTED_BYTE_LENGTH = 80;

    describe('connect:origins aggregateHash fold (VER-06)', () => {
      it('matches NUB-CONNECT §Conformance Fixture SHA-256 digest', () => {
        const { digest, byteLength } = foldConnectOrigins(FIXTURE_ORIGINS);
        expect(digest).toBe(EXPECTED_DIGEST);
        expect(byteLength).toBe(EXPECTED_BYTE_LENGTH);
      });

      it('removing one origin flips the digest (content-addressing)', () => {
        const subset = FIXTURE_ORIGINS.slice(0, 2);
        const { digest } = foldConnectOrigins(subset);
        expect(digest).not.toBe(EXPECTED_DIGEST);
      });

      it('adding one origin flips the digest', () => {
        const plus = [...FIXTURE_ORIGINS, 'https://new.example.com'];
        const { digest } = foldConnectOrigins(plus);
        expect(digest).not.toBe(EXPECTED_DIGEST);
      });

      it('input reordering (before sort) produces the same digest (sort is load-bearing)', () => {
        const reversed = [...FIXTURE_ORIGINS].reverse();
        const { digest } = foldConnectOrigins(reversed);
        expect(digest).toBe(EXPECTED_DIGEST);
      });

      it('casing change in an origin flips the digest (lowercase step is load-bearing)', () => {
        // Note: normalizeConnectOrigin rejects uppercase-host inputs, so this is a
        // "what if a shell doesn't pre-validate" defense test for the fold's own
        // lowercase step.
        const uppercased = [
          'HTTPS://API.EXAMPLE.COM',
          'https://xn--caf-dma.example.com',
          'wss://events.example.com',
        ];
        const { digest: upperFolded } = foldConnectOrigins(uppercased);
        // After the fold's lowercase step, the first origin becomes 'https://api.example.com'
        // which matches the fixture — so the digest IS equal.
        // BUT if we BYPASSED the lowercase step and only sorted/joined, the digest would differ.
        // This test asserts that the fold's internal lowercase step is applied:
        expect(upperFolded).toBe(EXPECTED_DIGEST);
      });

      it('empty origin list produces well-defined digest', () => {
        const { digest, byteLength } = foldConnectOrigins([]);
        expect(byteLength).toBe(0);
        // SHA-256 of empty string:
        expect(digest).toBe('e3b0c44298fc1c149afbf4c8996fb92427ae41e4649b934ca495991b7852b855');
      });

      it('content-addressing property: same dist + changed connect list → hash changes', () => {
        // Simulates VER-06 end-to-end scenario: napplet built twice, only the
        // connect:origins differ → aggregateHash value folded in from this output
        // MUST differ, invalidating any grant keyed on the prior (dTag, aggregateHash).
        const build1 = foldConnectOrigins(['https://api.example.com']);
        const build2 = foldConnectOrigins(['https://api.example.com', 'https://news.example.com']);
        expect(build1.digest).not.toBe(build2.digest);
      });
    });
    ```
