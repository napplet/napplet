  <action>
  In the `nappStorage.keys()` method (approximately lines 156-163), replace the current comma-split parsing:

  ```typescript
  async keys(): Promise<string[]> {
    const response = await sendStorageRequest('shell:storage-keys', []);
    // eslint-disable-next-line @typescript-eslint/no-explicit-any
    const event = response as any;
    const keysTag = event.tags?.find((t: string[]) => t[0] === 'keys');
    const raw = keysTag?.[1] ?? '';
    return raw === '' ? [] : raw.split(',');
  },
  ```

  With NIP tag parsing that filters for all `['key', ...]` tags:

  ```typescript
  async keys(): Promise<string[]> {
    const response = await sendStorageRequest('shell:storage-keys', []);
    // eslint-disable-next-line @typescript-eslint/no-explicit-any
    const event = response as any;
    const keyTags = event.tags?.filter((t: string[]) => t[0] === 'key') ?? [];
    return keyTags.map((t: string[]) => t[1]);
  },
  ```

  This correctly handles:
  - Keys containing commas (no delimiter collision)
  - Empty key lists (filter returns empty array)
  - Standard NIP tag format (consistent with the rest of the protocol)
