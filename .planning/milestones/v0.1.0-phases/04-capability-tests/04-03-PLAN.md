  ```typescript
  function buildStorageEvent(
    pubkey: string,
    topic: string,
    tags: string[][],
    correlationId: string,
  ): Record<string, unknown> {
    return {
      id: Math.random().toString(36).slice(2).padEnd(64, '0'),
      pubkey,
      created_at: Math.floor(Date.now() / 1000),
      kind: 29003, // BusKind.INTER_PANE
      tags: [['t', topic], ['id', correlationId], ...tags],
      content: '',
      sig: '0'.repeat(128),
    };
  }
  ```

  **Helper: inject storage request and get response:**
  ```typescript
  async function storageRequest(
    page: Page,
    windowId: string,
    topic: string,
    tags: string[][],
    correlationId: string,
    pubkey: string,
  ): Promise<TappedMessage> {
    const event = buildStorageEvent(pubkey, topic, tags, correlationId);
    await page.evaluate(([wid, evt]) => window.__publishEvent__(wid, evt), [windowId, event]);
    // Wait for response with matching correlation ID
    let response: TappedMessage | undefined;
    await expect.poll(async () => {
      const msgs = await page.evaluate(() => window.__TEST_MESSAGES__);
      response = msgs.find(m =>
        m.direction === 'shell->napplet' &&
        m.verb === 'EVENT' &&
        Array.isArray(m.raw[2]) === false &&
        typeof m.raw[2] === 'object' &&
        (m.raw[2] as any)?.tags?.some((t: string[]) => t[0] === 'id' && t[1] === correlationId)
      );
      return !!response;
    }, { timeout: 5000 }).toBe(true);
    return response!;
  }
  ```

  **Helper: extract tag value from response event:**
  ```typescript
  function getResponseTag(response: TappedMessage, tagName: string): string | undefined {
    const evt = response.raw[2] as { tags?: string[][] };
    return evt?.tags?.find((t: string[]) => t[0] === tagName)?.[1];
  }

  function getResponseTags(response: TappedMessage, tagName: string): string[] {
    const evt = response.raw[2] as { tags?: string[][] };
    return evt?.tags?.filter((t: string[]) => t[0] === tagName).map((t: string[]) => t[1]) ?? [];
  }
  ```

  Note: The napplet pubkey used in storage events MUST match the registered napplet pubkey from the AUTH handshake. Use `__getNappPubkey__(windowId)` or the pubkey from `__getNappEntry__(windowId)`.

  - Generate a value string larger than 512KB (524288 bytes). Use `'x'.repeat(524289)` — one byte over when combined with key. But careful: the quota calculation uses `TextEncoder().encode(scopedKey + value).length`. The scopedKey includes the prefix `napp-storage:${pubkey}:${dTag}:${aggregateHash}:${userKey}`. Need to account for scoped key length. A simpler approach: use a value of `'x'.repeat(600000)` which will definitely exceed 512KB including overhead.

  **STR-08: Quota accuracy — boundary test**
  - Calculate available bytes precisely. The quota is DEFAULT_STORAGE_QUOTA = 524288 bytes.
  - The scoped key format is `napp-storage:${pubkey}:${dTag}:${aggregateHash}:${userKey}`. Use a 1-char userKey like 'a' to minimize overhead.
  - Calculate scoped key byte length: `new TextEncoder().encode('napp-storage:' + pubkey + ':' + dTag + ':' + aggregateHash + ':a').length`
  - Since we can't easily calculate exact byte lengths in the test without knowing the runtime values, take a practical approach:
    1. First, write a value that uses ~500KB (well under quota) — should succeed
    2. Then write another key with remaining value that just barely pushes total over 524288 — should fail
  - Alternative practical approach:
    1. Write key 'fill' with value `'x'.repeat(524200)` — under quota, should succeed (OK true)
    2. Clear messages
    3. Write key 'overflow' with value `'x'.repeat(200)` — pushes total past 524288 (accounting for scoped key bytes), should fail with quota exceeded
  - Assert first write returns `['ok', 'true']`
  - Assert second write returns `['error', ...]` containing 'quota exceeded'
  - Test description: `test('STR-08: quota accuracy -- write at limit succeeds, over limit fails')`

  **STR-09: Storage persistence across reload**
  - Set a key: shell:storage-set key='persist-test' value='survive-reload'

  - Reload the page: `await page.reload()`
  - Wait for shell ready again: `await page.waitForFunction(() => window.__SHELL_READY__)`
  - Re-load auth-napplet and wait for AUTH OK
  - Get the new nappEntry (pubkey will be different after reload because auth-napplet generates a fresh keypair each time)
  - **Key insight:** After reload, the napplet gets a NEW ephemeral keypair. The old storage keys are scoped to the OLD pubkey. So "persistence" means the data is still in localStorage but the new napplet identity can't access it through the storage proxy because the scoped key is different.

    4. Wait for shell ready

    6. Read value directly: `await page.evaluate((key) => window.__getLocalStorageItem__(key), scopedKeyString)` — should return 'survive-reload'
  - To get the scoped key, read it from localStorage keys before reload — find the one starting with 'napp-storage:' that contains 'persist-test'
  - Test description: `test('STR-09: storage persistence -- values survive shell reload')`
