# Plan 16-01: Core Package Import Verification Tests

## Objective

Create unit tests for @napplet/core that prove all protocol types, constants, and message definitions are correctly re-exported from the package. These tests run in Node.js without a browser.

## Context

Phase 12 creates @napplet/core with all shared types and constants. TST-02 requires import verification tests proving the re-exports work. Since @napplet/core has zero dependencies, these are pure import-and-assert tests.

The root vitest.config.ts already includes `packages/*/src/**/*.test.ts` in its test glob, so tests placed in packages/core/src/ will be discovered automatically.

<action>
Create `packages/core/src/index.test.ts` that verifies every named export from @napplet/core.

The test file must:

1. Import all type exports and verify they are importable (TypeScript compile-time check):
   - `NostrEvent`, `NostrFilter`, `Capability`, `AclCheckEvent`
   - Any other protocol types moved to core in Phase 12

2. Import and assert all value exports are defined (runtime check):
   - `BusKind` — verify it has properties: `AUTH` (number), `SIGNER_REQUEST` (number), `SIGNER_RESPONSE` (number), `INTER_PANE` (number), `HOTKEY_FORWARD` (number)
   - `AUTH_KIND` — verify it equals `BusKind.AUTH` or is the correct number
   - `SHELL_BRIDGE_URI` — verify it is a non-empty string starting with `'shell://'`
   - `PROTOCOL_VERSION` — verify it is a non-empty string
   - `ALL_CAPABILITIES` — verify it is an array with length > 0 containing strings like `'relay:read'`, `'relay:write'`, `'sign:event'`, `'state:read'`, `'state:write'`
   - `DESTRUCTIVE_KINDS` — verify it is an array of numbers (kind 0, 3, 5, 10002)
   - `TOPICS` — verify it is an object with keys for state-get, state-set, state-keys, state-remove, state-clear, relay-scoped-connect, etc.

3. Verify no browser globals are referenced:
   - Import should succeed in Node.js `environment: 'node'` without polyfills for `window`, `document`, `localStorage`

Test structure:
```ts
import { describe, it, expect } from 'vitest';
import {
  BusKind,
  AUTH_KIND,
  SHELL_BRIDGE_URI,
  PROTOCOL_VERSION,
  ALL_CAPABILITIES,
  DESTRUCTIVE_KINDS,
  TOPICS,
} from './index.js';

// Type-level imports (compile check — if this file compiles, types are exported)
import type { NostrEvent, NostrFilter, Capability } from './index.js';

describe('@napplet/core exports', () => {
  describe('BusKind constants', () => {
    it('exports BusKind with all protocol kinds', () => {
      expect(BusKind).toBeDefined();
      expect(typeof BusKind.AUTH).toBe('number');
      expect(typeof BusKind.SIGNER_REQUEST).toBe('number');
      expect(typeof BusKind.SIGNER_RESPONSE).toBe('number');
      expect(typeof BusKind.INTER_PANE).toBe('number');
      expect(typeof BusKind.HOTKEY_FORWARD).toBe('number');
    });
  });

  describe('protocol constants', () => {
    it('exports AUTH_KIND as a number', () => {
      expect(typeof AUTH_KIND).toBe('number');
    });

    it('exports SHELL_BRIDGE_URI as shell:// URI', () => {
      expect(typeof SHELL_BRIDGE_URI).toBe('string');
      expect(SHELL_BRIDGE_URI).toMatch(/^shell:\/\//);
    });

    it('exports PROTOCOL_VERSION as non-empty string', () => {
      expect(typeof PROTOCOL_VERSION).toBe('string');
      expect(PROTOCOL_VERSION.length).toBeGreaterThan(0);
    });
  });

  describe('capability constants', () => {
    it('exports ALL_CAPABILITIES with known capabilities', () => {
      expect(Array.isArray(ALL_CAPABILITIES)).toBe(true);
      expect(ALL_CAPABILITIES.length).toBeGreaterThan(0);
      expect(ALL_CAPABILITIES).toContain('relay:read');
      expect(ALL_CAPABILITIES).toContain('relay:write');
      expect(ALL_CAPABILITIES).toContain('sign:event');
      expect(ALL_CAPABILITIES).toContain('state:read');
      expect(ALL_CAPABILITIES).toContain('state:write');
    });
  });

  describe('DESTRUCTIVE_KINDS', () => {
    it('exports DESTRUCTIVE_KINDS with metadata, contacts, relay list, nip46', () => {
      expect(Array.isArray(DESTRUCTIVE_KINDS)).toBe(true);
      expect(DESTRUCTIVE_KINDS).toContain(0);
      expect(DESTRUCTIVE_KINDS).toContain(3);
      expect(DESTRUCTIVE_KINDS).toContain(5);
      expect(DESTRUCTIVE_KINDS).toContain(10002);
    });
  });

  describe('TOPICS', () => {
    it('exports TOPICS object with shell command keys', () => {
      expect(typeof TOPICS).toBe('object');
      expect(TOPICS).not.toBeNull();
      // Verify known topic keys exist
      const topicValues = Object.values(TOPICS);
      expect(topicValues.length).toBeGreaterThan(0);
      // At least the state topics should exist
      expect(topicValues).toContain('shell:state-get');
      expect(topicValues).toContain('shell:state-set');
    });
  });

  describe('type-level exports (compile check)', () => {
    it('types are usable in type annotations', () => {
      // If this compiles, the types are exported correctly
      const _event: NostrEvent = {} as NostrEvent;
      const _filter: NostrFilter = {} as NostrFilter;
      const _cap: Capability = 'relay:read' as Capability;
      expect(true).toBe(true);
    });
  });
});
```
