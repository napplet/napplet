**Step 4 — Verify every exports entry has a matching dist file:**

```bash
node -e "
const fs = require('fs');
const path = require('path');
const pkg = require('./packages/nub/package.json');
const missing = [];
for (const [key, value] of Object.entries(pkg.exports)) {
  const importPath = path.join('packages/nub', value.import);
  const typesPath = path.join('packages/nub', value.types);
  if (!fs.existsSync(importPath)) missing.push(\`import missing: \${key} -> \${importPath}\`);
  if (!fs.existsSync(typesPath))  missing.push(\`types missing:  \${key} -> \${typesPath}\`);
}
if (missing.length) { console.error(missing.join('\n')); process.exit(1); }
console.log('All 36 export entries have paired .js + .d.ts on disk.');
"
```

Expected output: `All 36 export entries have paired .js + .d.ts on disk.`

**Step 5 — Verify root `.` import fails to resolve (EXP-04 at runtime):**

```bash
node -e "
try {
  const p = require.resolve('@napplet/nub', { paths: [process.cwd()] });
  console.error('FAIL: @napplet/nub root resolved to', p, '— EXP-04 violated');
  process.exit(1);
} catch (e) {
  if (e.code === 'ERR_PACKAGE_PATH_NOT_EXPORTED' || e.code === 'MODULE_NOT_FOUND') {
    console.log('OK: @napplet/nub root does not resolve —', e.code);
  } else {
    console.error('UNEXPECTED:', e);
    process.exit(1);
  }
}
" || true
```

The `|| true` is intentional — this test is treating a successful resolve as failure and a failed resolve as success; we want the "OK:" branch. Confirm the printed line starts with `OK:` and includes either `ERR_PACKAGE_PATH_NOT_EXPORTED` or `MODULE_NOT_FOUND`. If it starts with `FAIL:` or `UNEXPECTED:`, treat this step as failed and halt.

**Step 6 — Verify a domain subpath DOES resolve:**

```bash
node --input-type=module -e "import('@napplet/nub/relay/types').then(m => { if (m.DOMAIN !== 'relay') throw new Error('expected DOMAIN=relay, got '+m.DOMAIN); console.log('OK: @napplet/nub/relay/types resolved, DOMAIN=', m.DOMAIN); }).catch(e => { console.error(e); process.exit(1); });"
```

Expected: `OK: @napplet/nub/relay/types resolved, DOMAIN= relay`

**Step 7 — Verify types-only subpaths are free of @napplet/core runtime imports:**

```bash
for d in relay storage ifc keys theme media notify identity config; do
  if grep -l "require('@napplet/core')\|from '@napplet/core'" "packages/nub/dist/$d/types.js" 2>/dev/null; then
    echo "FAIL: packages/nub/dist/$d/types.js imports @napplet/core — types subpath must stay pure"
    exit 1
  fi
done
echo "OK: all 9 <domain>/types.js emits are free of @napplet/core runtime imports"
```

(Note: the source `types.ts` files only use `import type` from `@napplet/core`, which tsup strips at emit; this step confirms the emit is clean.)

**Step 8 — Verify 8 of 9 barrels carry registerNub + config does NOT:**

```bash

  echo "FAIL: config/index.js must not contain registerNub (side-effect-free barrel)"
  exit 1
fi
echo "OK: 8 barrels register, config barrel does not"
```
