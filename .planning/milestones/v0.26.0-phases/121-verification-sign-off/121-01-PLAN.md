// Types-only import from the relay subpath.
// Per @napplet/nub sideEffects:false + EXP-02 (granular subpath), this must

// and specifically no registerNub side effect from the relay barrel.

// Touch the type once so TS doesn't elide the import before esbuild sees it,
// while making sure the runtime value it "uses" is a pure string literal.
export const handler = (msg: RelaySubscribeMessage): string => {
  return msg.subId;
};
EOF

  --minify \
  --format=esm \
  --platform=neutral \
  --target=es2022 \
  src/consumer.ts \
  --outfile=bundle.js \
  2>&1 | tee esbuild.log

# ── Assertions ────────────────────────────────────────────────────────────
BUNDLE=/tmp/napplet-treeshake-verify/bundle.js
test -f "$BUNDLE"

# 1. Bundle size sanity: < 1024 bytes (type import vanishes; only `handler` remains).

test "$BYTES" -lt 1024

test "$REGISTER_COUNT" -eq 0

#    We grep for the *NUB domain string* pattern so we don't match the word
#    "config" inside JSON keys or harmless identifiers. The strings we care
#    about are the NubDomain literal strings that would appear if a barrel
#    was pulled in (e.g. the string `"identity"` in `registerNub("identity", ...)`).
#    For a <1KB bundle this is a pragmatic check — combined with assertion 2
#    (zero registerNub calls) it proves no cross-domain code leaked in.
CROSS_COUNT=0
for d in config identity ifc keys media notify storage theme; do
  HITS=$(grep -c -E "\"$d\"|'$d'" "$BUNDLE" || true)
  echo "cross_domain_hits[$d]=$HITS"
  CROSS_COUNT=$((CROSS_COUNT + HITS))
done
echo "cross_domain_total=$CROSS_COUNT"
test "$CROSS_COUNT" -eq 0

# Full bundle listing for SUMMARY.md (it's small; paste verbatim).
echo "── bundle.js (verbatim) ──"
cat "$BUNDLE"
echo "── end bundle.js ──"

  npm install --silent --no-audit --no-fund --prefer-offline > install.log 2>&1 || {
