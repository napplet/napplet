**src/types.ts** template (substitute {Domain}, {domain}, {description}):
```typescript
/**
 * @napplet/nub-{domain} -- {description} message types.
 *
 * This module defines the JSON envelope message types for the {domain} NUB domain.
 * All messages use the "domain.action" type discriminant format per NIP-5D.
 *
 * Message types will be populated from the NUB-{DOMAIN} spec (Phase 77).
 * For now, this establishes the base message interface that all {domain}
 * messages will extend.
 */

import type { NappletMessage } from '@napplet/core';

// ─── Domain Constants ──────────────────────────────────────────────────────

/** The NUB domain name for {domain} messages. */
export const DOMAIN = '{domain}' as const;

// ─── Base Message Type ─────────────────────────────────────────────────────

/**
 * Base interface for all {domain} NUB messages.
 * Extends NappletMessage with the {domain}.* type prefix.
 *
 * Concrete message types (e.g., {domain}.subscribe, {domain}.event)
 * will extend this interface with specific payload fields.
 * Those types are defined by the NUB-{DOMAIN} spec (Phase 77).
 *
 * @example
 * ```ts
 * // Future: concrete {domain} message type
 * // interface {Domain}SubscribeMessage extends {Domain}Message {
 * //   type: '{domain}.subscribe';
 * //   filters: NostrFilter[];
 * // }
 * ```
 */
export interface {Domain}Message extends NappletMessage {
  /** Message type in "{domain}.<action>" format. */
  type: `{domain}.${string}`;
}
```

Domain-specific substitutions:
- **relay**: Domain="Relay", domain="relay", DOMAIN description: "relay proxy operations (subscribe, publish, query)"
- **signer**: Domain="Signer", domain="signer", DOMAIN description: "signing delegation (sign, encrypt, decrypt)"
- **storage**: Domain="Storage", domain="storage", DOMAIN description: "scoped storage operations (get, set, remove, keys)"
- **ifc**: Domain="Ifc", domain="ifc", DOMAIN description: "inter-frame communication (dispatch, channel)"
