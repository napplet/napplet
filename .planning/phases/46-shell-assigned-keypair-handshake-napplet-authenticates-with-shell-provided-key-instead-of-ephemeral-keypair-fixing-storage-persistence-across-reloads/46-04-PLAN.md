# Plan 46-04: REGISTER/IDENTITY Handshake and Delegated Key Security

## Objective

Implement the complete REGISTER -> IDENTITY -> AUTH handshake flow. The runtime handles REGISTER messages by deriving a deterministic keypair from HMAC(shellSecret, dTag + aggregateHash), sends IDENTITY with the keypair to the napplet, and the napplet uses that keypair for standard NIP-42 AUTH. Add explicit guards preventing delegated key events from being published to external relays.

```typescript
/**
 * key-derivation.ts â€” Deterministic keypair derivation for napplet identity.
 *
 * Derives a stable secp256k1 keypair from HMAC-SHA256(shellSecret, dTag + aggregateHash).
 * Same napplet type + version + shell = same keypair.
 * No browser APIs. No DOM.
 */

import { hmac } from '@noble/hashes/hmac';
import { sha256 } from '@noble/hashes/sha256';
import type { ShellSecretPersistence } from './types.js';

/** 32-byte shell secret used as HMAC key for keypair derivation. */
const SECRET_LENGTH = 32;

/**
 * Get or create the shell secret.
 * Generated once on first use, persisted via ShellSecretPersistence.
 *
 * @param persistence - Storage backend for the shell secret
 * @param randomBytes - Function to generate cryptographically secure random bytes
 * @returns The 32-byte shell secret
 */
export function getOrCreateShellSecret(
  persistence: ShellSecretPersistence,
  randomBytes: (length: number) => Uint8Array,
): Uint8Array {
  const existing = persistence.get();
  if (existing && existing.length === SECRET_LENGTH) return existing;

  const secret = randomBytes(SECRET_LENGTH);
  persistence.set(secret);
  return secret;
}

/**
 * Derive a deterministic private key seed from HMAC-SHA256(shellSecret, dTag + aggregateHash).
 *
 * The output is 32 bytes â€” a valid secp256k1 private key with overwhelming probability
 * (failure probability ~3.7 * 10^-39, less than a hardware error).
 *
 * @param shellSecret - The 32-byte per-shell secret
 * @param dTag - Napplet type identifier
 * @param aggregateHash - Build artifact hash (empty string in dev mode)
 * @returns 32-byte seed suitable for use as a secp256k1 private key
 */
export function derivePrivateKey(
  shellSecret: Uint8Array,
  dTag: string,
  aggregateHash: string,
): Uint8Array {
  const input = new TextEncoder().encode(dTag + aggregateHash);
  return hmac(sha256, shellSecret, input);
}
```

**Important:** `@noble/hashes` is a transitive dependency of `nostr-tools` (which is a peer dep of @napplet/runtime). The imports `@noble/hashes/hmac` and `@noble/hashes/sha256` resolve through the existing dependency tree â€” no new dependencies needed. However, `@noble/hashes` must be listed as a dependency in `packages/runtime/package.json` to ensure it resolves correctly when runtime is consumed standalone.

Add `@noble/hashes` to `packages/runtime/package.json` dependencies:
```json

  "@noble/hashes": "^1.3.0"
}
```

If `packages/runtime/package.json` already has a `dependencies` field, add `@noble/hashes` to it. If not, create the field. Then run `pnpm install` to update the lockfile.

```typescript
  async function handleRegister(msg: unknown[], windowId: string): Promise<void> {
    const payload = msg[1] as RegisterPayload | undefined;
    if (!payload || typeof payload !== 'object') {
      hooks.sendToNapplet(windowId, ['NOTICE', 'invalid REGISTER payload']);
      return;
    }

    const { dTag, claimedHash } = payload;
    if (!dTag || typeof dTag !== 'string') {
      hooks.sendToNapplet(windowId, ['NOTICE', 'REGISTER requires dTag']);
      return;
    }

    const aggregateHash = (typeof claimedHash === 'string') ? claimedHash : '';

    // Derive deterministic keypair
    if (!shellSecret) {
      hooks.sendToNapplet(windowId, ['NOTICE', 'shell secret not available â€” cannot derive keypair']);
      return;
    }

    const privkeyBytes = derivePrivateKey(shellSecret, dTag, aggregateHash);
    const pubkey = getPublicKey(privkeyBytes);
    const privkeyHex = bytesToHex(privkeyBytes);

    // Assign or retrieve persistent GUID for this iframe
    let instanceId: string;
    if (hooks.guidPersistence) {
      const existing = hooks.guidPersistence.get(windowId);
      if (existing) {
        instanceId = existing;
      } else {
        instanceId = hooks.crypto.randomUUID();
        hooks.guidPersistence.set(windowId, instanceId);
      }
    } else {
      instanceId = hooks.crypto.randomUUID();
    }

    // Store registration state for the upcoming AUTH
    pendingRegistrations.set(windowId, {
      dTag, aggregateHash, pubkey, instanceId,
    });

    // Send IDENTITY to napplet
    const identityPayload: IdentityPayload = {
      pubkey,
      privkey: privkeyHex,
      dTag,
      aggregateHash,
    };
    hooks.sendToNapplet(windowId, [VERB_IDENTITY, identityPayload]);

    // Now send the AUTH challenge (the napplet will sign with the delegated key)
    runtimeInstance.sendChallenge(windowId);
  }
```
