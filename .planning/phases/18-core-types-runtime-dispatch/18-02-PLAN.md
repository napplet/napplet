## 1. ServiceHandler and ServiceRegistry in runtime/src/types.ts

Add after the RuntimeHooks interface (after line 376). Import ServiceDescriptor from @napplet/core.

```typescript
// ─── Service Extension Types ──────────────────────────────────────────────

/**
 * Handler for service-specific messages from napplets.
 * The runtime dispatches INTER_PANE events to the appropriate handler
 * based on the service name extracted from the topic prefix.
 *
 * @example
 * ```ts
 * const handler: ServiceHandler = {
 *   descriptor: { name: 'audio', version: '1.0.0' },
 *   handleRequest(windowId, topic, content, event) {
 *     if (topic === 'audio:register') {
 *       audioManager.register(windowId, content.nappClass, content.title);
 *     }
 *   },
 * };
 * ```
 */
export interface ServiceHandler {
  /** Metadata describing this service. */
  descriptor: ServiceDescriptor;
  /**
   * Handle a service request from a napplet.
   *
   * @param windowId - The requesting napplet's window identifier
   * @param topic - The full topic string (e.g., 'audio:register')
   * @param content - Parsed JSON content from the event
   * @param event - The raw NostrEvent for advanced use cases
   */
  handleRequest(windowId: string, topic: string, content: unknown, event: NostrEvent): void;
  /**
   * Called when a napplet window is destroyed. Services should clean up
   * any state associated with the window.
   *
   * @param windowId - The destroyed napplet's window identifier
   */
  onWindowDestroyed?(windowId: string): void;
}

/**
 * Registry of service handlers, keyed by service name.
 *
 * @example
 * ```ts
 * const services: ServiceRegistry = {
 *   audio: audioHandler,
 *   notifications: notificationHandler,
 * };
 * ```
 */
export interface ServiceRegistry {
  [serviceName: string]: ServiceHandler;
}
```

The import for ServiceDescriptor goes at the top of types.ts alongside NostrEvent:

```typescript
import type { NostrEvent, ServiceDescriptor } from '@napplet/core';
```

Note: types.ts currently uses `import type { NostrEvent } from '@napplet/core';` — just add ServiceDescriptor to that import.

## 5. routeToService helper function

Add this function inside createRuntime(), near the top of the function body (after module-level state declarations):

```typescript
  /** Route an INTER_PANE topic to a registered service handler. Returns true if handled. */
  function routeToService(windowId: string, topic: string, event: NostrEvent): boolean {
    const colonIndex = topic.indexOf(':');
    if (colonIndex === -1) return false;
    const prefix = topic.slice(0, colonIndex);
    const handler = services[prefix];
    if (!handler) return false;

    let content: unknown;
    try { content = event.content ? JSON.parse(event.content) : {}; }
    catch { content = {}; }

    handler.handleRequest(windowId, topic, content, event);
    return true;
  }
```
