  Create `tests/e2e/inter-pane.spec.ts` following established test patterns.

  **Critical routing knowledge:** Inter-pane events are kind `29003` (BusKind.INTER_PANE) with a `['t', topic]` tag. Topics that do NOT start with `shell:` are routed through `storeAndRoute()` which calls `deliverToSubscriptions()`. The sender is excluded from delivery (`senderId !== null && sub.windowId === senderId`).

  For a napplet to receive inter-pane events, it must have an active REQ subscription with a filter matching kind 29003 and optionally `#t: [topic]`.

  **Multi-napplet setup pattern:**
  Loading multiple auth-napplets produces multiple independent identities. Each auth-napplet generates a fresh ephemeral Ed25519 keypair, so each gets a unique pubkey.

  **Helper: load napplet and wait for AUTH:**
  ```typescript
  async function loadAndAuth(page: Page, label: string): Promise<{ windowId: string; pubkey: string }> {
    const windowId = await page.evaluate(() => (window as any).__loadNapplet__('auth-napplet'));
    // Wait for this napplet to complete AUTH -- check for OK true count
    const existingOks = await page.evaluate(() =>
      (window as any).__TEST_MESSAGES__.filter(
        (m: any) => m.verb === 'OK' && m.parsed.success === true
      ).length
    );
    await expect.poll(async () => {
      const msgs = await page.evaluate(() => (window as any).__TEST_MESSAGES__);
      const oks = msgs.filter((m: any) => m.verb === 'OK' && m.parsed.success === true);
      return oks.length;
    }, { timeout: 15000 }).toBeGreaterThan(existingOks);

    const pubkey = await page.evaluate((wid) => (window as any).__getNappPubkey__(wid), windowId);
    return { windowId, pubkey };
  }
  ```

  **Helper: create topic subscription for a napplet:**
  ```typescript
  async function subscribeTopic(page: Page, windowId: string, subId: string, topic: string): Promise<void> {
    await page.evaluate(
      ([wid, sid, t]) => (window as any).__createSubscription__(wid, sid, [{ kinds: [29003], '#t': [t] }]),
      [windowId, subId, topic]
    );
  }
  ```

  **Helper: emit an inter-pane event from a napplet:**
  ```typescript
  function buildIpcEvent(pubkey: string, topic: string, content: string): Record<string, unknown> {
    return {
      id: Math.random().toString(36).slice(2).padEnd(64, '0'),
      pubkey,
      created_at: Math.floor(Date.now() / 1000),
      kind: 29003, // BusKind.INTER_PANE
      tags: [['t', topic]],
      content,
      sig: '0'.repeat(128),
    };
  }

  async function emitIpc(page: Page, windowId: string, pubkey: string, topic: string, content: string): Promise<void> {
    const event = buildIpcEvent(pubkey, topic, content);
    await page.evaluate(([wid, evt]) => (window as any).__publishEvent__(wid, evt), [windowId, event]);
  }
  ```

  **Helper: check if a napplet received an EVENT with a specific topic:**
  ```typescript
  function findIpcDelivery(
    messages: TappedMessage[],
    topic: string,
    afterIndex: number = 0,
  ): TappedMessage | undefined {
    return messages.find((m, idx) =>
      idx >= afterIndex &&
      m.verb === 'EVENT' &&
      m.direction === 'shell->napplet' &&
      typeof m.raw[2] === 'object' &&
      (m.raw[2] as any)?.kind === 29003 &&
      (m.raw[2] as any)?.tags?.some((t: string[]) => t[0] === 't' && t[1] === topic)
    );
  }
  ```

  **Test setup (beforeEach):**
  1. Navigate to harness page: `await page.goto('/')`
  2. Wait for shell ready: `await page.waitForFunction(() => window.__SHELL_READY__)`
  3. Clear ACL state: `await page.evaluate(() => window.__aclClear__())`
  4. Clear messages: `await page.evaluate(() => window.__clearMessages__())`

  Note: napplet loading is done per-test because different tests need different numbers of napplets (2 or 3).
