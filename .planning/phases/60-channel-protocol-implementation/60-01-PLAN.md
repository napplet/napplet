/**
 * NIP-01 verb for opening a pipe to a target napplet (napplet -> shell).
 * Payload: { targetType: string, targetKey?: string, name?: string }
 */
export const VERB_PIPE_OPEN = 'PIPE_OPEN' as const;

 * Payload: { pipeId: string, peer: { pubkey: string, dTag: string } }

export const VERB_PIPE_ACK = 'PIPE_ACK' as const;

/**
 * NIP-01 verb for pipe data messages (bidirectional).
 * Format: ["PIPE", pipeId, payload] — payload is opaque to the shell.
 */
export const VERB_PIPE = 'PIPE' as const;

/**
 * NIP-01 verb for closing a pipe (bidirectional).
 * Format: ["PIPE_CLOSE", pipeId, reason?]
 */
export const VERB_PIPE_CLOSE = 'PIPE_CLOSE' as const;

/**
 * NIP-01 verb for broadcast to all pipe peers (napplet -> shell).
 * Format: ["PIPE_BROADCAST", payload] — shell fans out as individual PIPE messages.

Add the following types to `packages/core/src/types.ts` after the `ServiceInfo` type and before the `NappletGlobal` interface:

```typescript
// ─── Pipe Types ──────────────────────────────────────────────────────────

/**
 * Options for opening a pipe to a target napplet.
 *
 * @example
 * ```ts
 * const pipe = await window.napplet.pipes.open('synth-plugin', { targetKey: 'abc...' });
 * ```
 */
export interface PipeOpenOptions {
  /** Session pubkey of a specific target instance (optional — shell picks if omitted). */
  targetKey?: string;
  /** Named pipe for multiplexing multiple logical pipes to the same target. */
  name?: string;
}

/**
 * Identity of a connected pipe peer.
 */
export interface PipePeer {
  /** Peer's session public key. */
  pubkey: string;
  /** Peer's napplet type (dTag from manifest). */
  dTag: string;
}

/**
 * Handle for an open pipe connection. WebSocket-like API.
 *
 * @example
 * ```ts
 * const pipe = await window.napplet.pipes.open('synth-plugin');
 * pipe.on((payload) => console.log('received:', payload));
 * pipe.send({ bpm: 120 });
 * pipe.onClose(() => console.log('pipe closed'));
 * pipe.close();
 * ```
 */
export interface PipeHandle {
  /** Shell-assigned opaque pipe identifier. */
  readonly id: string;
  /** Connected peer identity. */
  readonly peer: PipePeer;
  /** Send data to the connected peer. Payload is opaque — any JSON-serializable value. */
  send(payload: unknown): void;
  /** Listen for incoming data from the peer. Returns a handle to stop listening. */
  on(callback: (payload: unknown) => void): { close(): void };
  /** Close the pipe. Notifies the peer via PIPE_CLOSE. */
  close(): void;
  /** Listen for pipe close notification (from peer or shell). */
  onClose(callback: (reason?: string) => void): void;
}

/**
 * Subscription handle for incoming pipe connections.
 *
 * @example
 * ```ts
 * const listener = window.napplet.pipes.onOpen((pipe) => {
 *   console.log('incoming pipe from:', pipe.peer.dTag);
 *   pipe.on((data) => console.log('data:', data));
 * });
 * // Later: listener.close();
 * ```
 */
export interface PipeOpenListener {
  /** Stop listening for incoming pipes. */
  close(): void;
}
```

Add `pipes` property to the `NappletGlobal` interface, after the `storage` property:

```typescript
  /**
   * Pipe protocol: point-to-point authenticated connections between napplets.
   * Shell-mediated — all data flows through the shell for routing and auth.
   */
  pipes: {
    /**
     * Open a pipe to a target napplet type.
     * @param targetType  dTag of the target napplet (from manifest)
     * @param options     Optional: targetKey for specific instance, name for multiplexing
     * @returns Promise resolving to a PipeHandle when the pipe is established
     */
    open(targetType: string, options?: PipeOpenOptions): Promise<PipeHandle>;
    /**
     * Listen for incoming pipe connections from other napplets.
     * @param callback  Called with a PipeHandle for each incoming connection
     * @returns A handle to stop listening
     */
    onOpen(callback: (pipe: PipeHandle) => void): PipeOpenListener;
    /**
     * Broadcast data to all connected pipe peers. Fire-and-forget.
     * Shell delivers as individual PIPE messages to each peer (sender excluded).
     * @param payload  Any JSON-serializable value
     */
    broadcast(payload: unknown): void;
  };
```
