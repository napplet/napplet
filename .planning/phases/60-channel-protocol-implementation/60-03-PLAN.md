```typescript
/**
 * pipes-shim.ts — Pipe client API for inter-napplet communication.
 *
 * Installs window.napplet.pipes with open(), onOpen(), and broadcast().
 * Uses postMessage to communicate with the shell runtime via PIPE_* verbs.
 * PIPE_OPEN -> PIPE_ACK correlation uses the same UUID + pending Map
 * pattern as nipdb-shim.ts.
 */

import {
  VERB_PIPE_OPEN,
  VERB_PIPE_ACK,
  VERB_PIPE,
  VERB_PIPE_CLOSE,
  VERB_PIPE_BROADCAST,
} from '@napplet/core';
import type { PipeHandle, PipeOpenOptions, PipePeer, PipeOpenListener } from '@napplet/core';
```

**Module-level state:**

```typescript
/** Timeout for PIPE_OPEN -> PIPE_ACK correlation (10 seconds). */
const PIPE_OPEN_TIMEOUT_MS = 10_000;

/** Pending PIPE_OPEN requests awaiting PIPE_ACK. Keyed by a local correlationId. */
const pendingOpens = new Map<string, {
  resolve: (pipe: PipeHandle) => void;
  reject: (reason: Error) => void;
  targetType: string;
}>();

/** Active pipe data listeners: pipeId -> Set of callbacks. */
const pipeDataListeners = new Map<string, Set<(payload: unknown) => void>>();

/** Active pipe close listeners: pipeId -> Set of callbacks. */
const pipeCloseListeners = new Map<string, Set<(reason?: string) => void>>();

/** Incoming pipe open listeners. */
const onOpenListeners = new Set<(pipe: PipeHandle) => void>();

/** Set of known pipeIds (received via PIPE_ACK) for this napplet. */
const activePipes = new Set<string>();
```

**PipeHandle factory function:**

```typescript
function createPipeHandle(pipeId: string, peer: PipePeer): PipeHandle {
  return {
    get id() { return pipeId; },
    get peer() { return peer; },

    send(payload: unknown): void {
      if (!activePipes.has(pipeId)) return; // Pipe already closed
      window.parent.postMessage([VERB_PIPE, pipeId, payload], '*');
    },

    on(callback: (payload: unknown) => void): { close(): void } {
      let listeners = pipeDataListeners.get(pipeId);
      if (!listeners) {
        listeners = new Set();
        pipeDataListeners.set(pipeId, listeners);
      }
      listeners.add(callback);
      return {
        close() { listeners!.delete(callback); },
      };
    },

    close(): void {
      if (!activePipes.has(pipeId)) return;
      activePipes.delete(pipeId);
      window.parent.postMessage([VERB_PIPE_CLOSE, pipeId], '*');
      // Clean up listeners
      pipeDataListeners.delete(pipeId);
      pipeCloseListeners.delete(pipeId);
    },

    onClose(callback: (reason?: string) => void): void {
      let listeners = pipeCloseListeners.get(pipeId);
      if (!listeners) {
        listeners = new Set();
        pipeCloseListeners.set(pipeId, listeners);
      }
      listeners.add(callback);
    },
  };
}
```

**Message handler:**

```typescript
function handlePipeMessage(event: MessageEvent): void {
  if (event.source !== window.parent) return;
  const msg = event.data;
  if (!Array.isArray(msg) || msg.length < 2) return;

  const verb = msg[0];

  switch (verb) {
    case VERB_PIPE_ACK: {
      const info = msg[1] as { pipeId: string; peer: PipePeer } | undefined;
      if (!info?.pipeId) return;
      const { pipeId, peer } = info;
      activePipes.add(pipeId);

      // Check if this is a response to our PIPE_OPEN
      // We match by: we have a pending open, and no pipe with this id yet
      // The shell sends ACK to both endpoints — only the opener has a pendingOpen
      let matched = false;
      for (const [corrId, pending] of pendingOpens.entries()) {
        // First pending open that matches — shell only allows one outstanding
        const handle = createPipeHandle(pipeId, peer);
        pendingOpens.delete(corrId);
        pending.resolve(handle);
        matched = true;
        break;
      }

      if (!matched) {
        // This is an incoming pipe (we are the target, not the opener)
        const handle = createPipeHandle(pipeId, peer);
        for (const listener of onOpenListeners) {
          try { listener(handle); } catch { /* listener error — best effort */ }
        }
      }
      break;
    }

    case VERB_PIPE: {
      const pipeId = msg[1] as string;
      const payload = msg[2];
      if (!pipeId || !activePipes.has(pipeId)) return;

      const listeners = pipeDataListeners.get(pipeId);
      if (listeners) {
        for (const cb of listeners) {
          try { cb(payload); } catch { /* listener error — best effort */ }
        }
      }
      break;
    }

    case VERB_PIPE_CLOSE: {
      const pipeId = msg[1] as string | null;
      const reason = (msg[2] as string) || undefined;

      if (pipeId === null) {
        // Error during PIPE_OPEN (before pipeId assignment)
        // Reject the first pending open
        for (const [corrId, pending] of pendingOpens.entries()) {
          pendingOpens.delete(corrId);
          pending.reject(new Error(reason ?? 'pipe open failed'));
          break;
        }
        return;
      }

      if (!activePipes.has(pipeId)) return;
      activePipes.delete(pipeId);

      const closeListeners = pipeCloseListeners.get(pipeId);
      if (closeListeners) {
        for (const cb of closeListeners) {
          try { cb(reason); } catch { /* listener error — best effort */ }
        }
      }

      pipeDataListeners.delete(pipeId);
      pipeCloseListeners.delete(pipeId);
      break;
    }
  }
}
```

**Public API functions:**

```typescript
function open(targetType: string, options?: PipeOpenOptions): Promise<PipeHandle> {
  return new Promise<PipeHandle>((resolve, reject) => {
    const corrId = crypto.randomUUID();

    pendingOpens.set(corrId, { resolve, reject, targetType });

    const openPayload: Record<string, unknown> = { targetType };
    if (options?.targetKey) openPayload.targetKey = options.targetKey;
    if (options?.name) openPayload.name = options.name;

    window.parent.postMessage([VERB_PIPE_OPEN, openPayload], '*');

    setTimeout(() => {
      if (pendingOpens.delete(corrId)) {
        reject(new Error('PIPE_OPEN timed out'));
      }
    }, PIPE_OPEN_TIMEOUT_MS);
  });
}

function onOpen(callback: (pipe: PipeHandle) => void): PipeOpenListener {
  onOpenListeners.add(callback);
  return {
    close() { onOpenListeners.delete(callback); },
  };
}

function broadcast(payload: unknown): void {
  window.parent.postMessage([VERB_PIPE_BROADCAST, payload], '*');
}
```

**Install function:**

```typescript
/**
 * Install the pipes message handler.
 * Called from index.ts during shim initialization.
 * @returns cleanup function
 */
export function installPipesShim(): () => void {
  window.addEventListener('message', handlePipeMessage);

  return () => {
    window.removeEventListener('message', handlePipeMessage);
    pendingOpens.clear();
    pipeDataListeners.clear();
    pipeCloseListeners.clear();
    onOpenListeners.clear();
    activePipes.clear();
  };
}

/** @internal Used by index.ts to wire window.napplet.pipes. */
export const pipesApi = { open, onOpen, broadcast };
```
