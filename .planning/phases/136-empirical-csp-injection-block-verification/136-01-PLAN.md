    ```javascript
    // /tmp/napplet-136-injection-block.cjs
    // Phase 136 VER-04 / DETECT-01 empirical gate.
    // Proves NUB-CLASS-1 nonce-based script-src blocks legacy <script>-tag injection
    // AND fires securitypolicyviolation observable from page-side JS.
    const { chromium } = require('/usr/lib/node_modules/playwright');

    (async () => {
      const NONCE = 'napplet136';
      // NUB-CLASS-1 10-directive baseline matching packages/vite-plugin/src/csp.ts prod output.
      // Note: meta-delivered CSP cannot include report-to (W3C CSP3 §4.2 header-only directive);
      // the page-side securitypolicyviolation event is the observable we use here.
      const csp = [
        "default-src 'none'",
        `script-src 'nonce-${NONCE}' 'self'`,
        "connect-src 'none'",
        "img-src blob: data:",
        "font-src blob: data:",
        "style-src 'self'",
        "worker-src 'none'",
        "object-src 'none'",
        "base-uri 'none'",
        "form-action 'none'",
      ].join('; ');

      const html =
        `<!DOCTYPE html><html><head>` +
        `<meta http-equiv="Content-Security-Policy" content="${csp}">` +
        `<script nonce="${NONCE}">` +
        `window.__violations = [];` +
        `document.addEventListener('securitypolicyviolation', (e) => {` +
        `  window.__violations.push({` +
        `    violatedDirective: e.violatedDirective,` +
        `    effectiveDirective: e.effectiveDirective,` +
        `    blockedURI: e.blockedURI,` +
        `    documentURI: e.documentURI,` +
        `    sourceFile: e.sourceFile,` +
        `    lineNumber: e.lineNumber,` +
        `    statusCode: e.statusCode,` +
        `  });` +
        `});` +
        `</script>` +
        `</head><body>phase-136-injection-fixture</body></html>`;

      const dataUrl = 'data:text/html;charset=utf-8,' + encodeURIComponent(html);

      const browser = await chromium.launch({
        headless: true,
        args: ['--ozone-platform=wayland'],
      });
      const ctx = await browser.newContext();
      const page = await ctx.newPage();

      let cspViolationFromConsole = false;
      page.on('console', (m) => {
        if (m.type() === 'error' && /Content Security Policy/i.test(m.text())) {
          cspViolationFromConsole = true;
        }
      });

      await page.goto(dataUrl, { waitUntil: 'load' }).catch(() => {});
      await page.waitForTimeout(200);

      // Simulate legacy NIP-07 content-script injection (nonce absent on the element).
      await page.evaluate(() => {
        const s = document.createElement('script');
        s.textContent = 'window.nostr = { __injected: true };';
        // No .nonce / no nonce attribute — simulating a content-script from an extension
        // that cannot know the per-page nonce.
        document.head.appendChild(s);
      });
      await page.waitForTimeout(200);

      const snapshot = await page.evaluate(() => ({
        windowNostrDefined: typeof window.nostr !== 'undefined',
        violations: window.__violations,
      }));

      const firstViolation = (snapshot.violations && snapshot.violations[0]) || {};
      const violatedDirective = firstViolation.violatedDirective || null;
      const effectiveDirective = firstViolation.effectiveDirective || null;

      const pass =
        cspViolationFromConsole &&
        Array.isArray(snapshot.violations) &&
        snapshot.violations.length >= 1 &&
        typeof violatedDirective === 'string' &&
        violatedDirective.startsWith('script-src') &&
        snapshot.windowNostrDefined === false;

      // stdout: the pass/fail summary consumed by /tmp/napplet-136-injection-block.log
      console.log(
        JSON.stringify({
          cspViolation: cspViolationFromConsole,
          windowNostrDefined: snapshot.windowNostrDefined,
          violationCount: snapshot.violations.length,
          violatedDirective,
          effectiveDirective,
          pass,
        }),
      );

      // stderr: the 4-field report shape consumed by /tmp/napplet-136-report-shape.log
      console.error(
        JSON.stringify({
          reportShape: {
            violatedDirective: firstViolation.violatedDirective || null,
            blockedURI: firstViolation.blockedURI || null,
            documentURI: firstViolation.documentURI || null,
            sourceFile: firstViolation.sourceFile || null,
          },
        }),
      );

      await browser.close();
      process.exit(pass ? 0 : 1);
    })().catch((err) => {
      console.error(JSON.stringify({ fixtureError: String(err && err.stack || err) }));
      process.exit(2);
    });
    ```
