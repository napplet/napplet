<task type="human">
  <name>Task 3: Publish @napplet/shim to npm</name>
  <files></files>
  <read_first>packages/shim/package.json</read_first>
  <action>
  Publish the shim package:

  ```bash
  cd packages/shim && npm publish --tag alpha --access public
  ```

  Verify publication:
  ```bash
  npm view @napplet/shim@alpha version
  ```

  Expected output: `0.1.0-alpha.1`

  If publish fails:
  - `403 Forbidden`: Scope not available or not authorized. Check npm org settings.
  - `402 Payment Required`: Scoped packages require npm organization. Create `@napplet` org on npmjs.com.
  - Other errors: Read error message, fix, and retry.
  </action>
  <verify>npm view @napplet/shim@alpha version 2>/dev/null | grep "0.1.0-alpha.1"</verify>
  <acceptance_criteria>
    - `npm publish` exits 0 for @napplet/shim
    - `npm view @napplet/shim@alpha version` returns "0.1.0-alpha.1"
    - Package is publicly accessible on npmjs.com
  </acceptance_criteria>
  <done>@napplet/shim published to npm</done>
</task>
