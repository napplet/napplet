```bash
  esac
  # Add file: dep
  node -e "
    const fs = require('fs');
    const p = JSON.parse(fs.readFileSync('package.json','utf8'));
    p.dependencies = p.dependencies || {};

const domains = ['config','identity','ifc','keys','media','notify','relay','storage','theme'];
let failed = 0;
for (const d of domains) {
  const old = await import(`@napplet/nub-${d}`);
  const neu = await import(`@napplet/nub/${d}`);
  const oldKeys = Object.keys(old).sort();
  const newKeys = Object.keys(neu).sort();
  const match = JSON.stringify(oldKeys) === JSON.stringify(newKeys);
  console.log(`[${d}] old keys: ${oldKeys.length}, new keys: ${newKeys.length}, match: ${match}`);
  if (!match) {
    console.error(`  DIFF old-only: ${oldKeys.filter(k => !newKeys.includes(k)).join(',')}`);
    console.error(`  DIFF new-only: ${newKeys.filter(k => !oldKeys.includes(k)).join(',')}`);
    failed++;
  }
}
if (failed) { console.error(`FAIL: ${failed}/9 domains have export shape mismatch`); process.exit(1); }
console.log('OK: all 9 deprecated packages have identical export surface to their @napplet/nub/<domain> canonical barrel');
EOF

node smoke.mjs
```
